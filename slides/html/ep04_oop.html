Let's create one to describe a customer

a class-y customer


class Customer:
    """
    Data structure to describe a customer with the business
    """
    def __init__(self, name, deposit):
        # customer attributes
        self.name = name
        self.deposit = deposit

    def customer_detail(self):
        """
        Print current financial position of customer
        """
        message = "Customer: %s has $%i" % (self.name, self.deposit)
        print(message)

class Customer:
    """
    Data structure to describe a customer for the business
    """
    _num_customers = 0
    
    def __init__(self, name, deposit):
        # keep track of customers
        Customer._num_customers += 1

        # attributes
        self.name = name
        self.accnum = Customer._num_customers
        self.deposit = deposit

    def compound_interest(self, rate, freq, term):
        """
        calculates compound interest
        """
        P_new = self.deposit*(1 + rate/freq)**(freq*term)

        return P_new
